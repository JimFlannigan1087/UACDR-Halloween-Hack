ACTOR RelicHighJump : UACDRInvBase 13370
{
	Inventory.PickupMessage "Picked up High Jump Relic"
	States
	{
	Spawn:
		HIJR ABCD 6
		Loop
	Pickup:
		HIJR A 0 ACS_NamedExecuteAlways("getHighJump")
		HIJR A 1 A_GiveInventory("jimsexHighJump",1,0)
		Stop
	}
}
