
ACTOR MonaShotgun : Weapon
{
  Weapon.SelectionOrder 1300
  Weapon.AmmoUse 1
  Weapon.AmmoGive 8
  Weapon.AmmoType "Shell"
  Weapon.SlotNumber 3
  +WEAPON.NOAUTOFIRE
  Inventory.PickupMessage "You got the shotgun! Lock and load!"
  Obituary "%o was blown open by %k's shotgun."
  Attacksound "MonaShotgun/Fire"
  Decal "BulletChip"
  States
  {
  Ready:
    DNSG A 1 A_WeaponReady
    Loop
  Deselect:
	TNT1 AAAA 0 A_Lower
    DNSG A 1 A_Lower
    Loop
  Select:
	TNT1 AAAA 0 A_Raise
    DNSG A 1 A_Raise
    Loop
  Fire:
    DNSG A 3
    DNSG B 0 A_FireBullets (6, 4, 10, 5, "BulletPuff")
    DNSG B 0 radius_quake(2, 2, 0, 1, 0)
    DNSG B 1 A_GunFlash
    DNSG CDEDCB 1 bright
    DNSG A 2
    DNSG FGH 1
    DNSG I 2 A_PlaySound ("MonaShotgun/Pump")
    DNSG JKLKJI 2
    DNSG HGF 1
    DNSG A 2
    Goto Ready
  Flash:
    TNT1 A 4 Bright A_Light1
    TNT1 A 3 Bright A_Light2
    Goto LightDone
  Spawn:
    DNSP A -1
    Stop
  }
}	

ACTOR MonaSSG : Weapon
{  
	Weapon.SelectionOrder 1300
  Weapon.AmmoUse 2
  Weapon.AmmoGive 16
  Weapon.AmmoType "Shell"
  +WEAPON.NOAUTOFIRE
  Inventory.PickupMessage "You got the Super Shotgun! This is gonna hurt!"
  Obituary "%o was splattered by %k's super shotgun."
  Attacksound "MonaSSG/Fire"
   Decal "BulletChip"
   States
   {
   Ready:
      SSG3 A 1 A_WeaponReady
      Loop
   Select:
		TNT1 AAA 0 A_Raise
      SSG3 A 1 A_Raise
      Loop
   Deselect:
		TNT1 AAA 0 A_Lower
      SSG3 A 1 A_Lower
      Loop
   Fire:
      SSG3 K 0 Bright A_FireBullets (11.2, 7.1, 30, 5, "BulletPuff")
	  SSG3 K 0 radius_quake(5, 3, 0, 1, 0)
	  SSG3 K 0 A_PlaySound ("MonaSSG/Fire", CHAN_WEAPON)
	  SSG3 K 1 A_GunFlash
	  SSG3 K 3
      SSG3 L 4
      SSG3 A 4 A_CheckReload
      SSG3 B 4
      SSG3 C 4 A_PlaySound ("MonaSSG/Open", CHAN_WEAPON)
      SSG3 D 8
      SSG3 EFG 2
      SSG3 H 3 A_PlaySound ("MonaSSG/Load", CHAN_WEAPON)
      SSG3 D 4
      SSG3 I 2 A_PlaySound ("MonaSSG/Close", CHAN_WEAPON)
      SSG3 A 2
      Goto Ready
   Flash:
      TNT1 A 1 A_Light1
      TNT1 A 1 A_Light2
      TNT1 A 1 A_Light0
      Stop
  Spawn:
    SGN2 A -1
    Stop
   }
}
